    PRIVATE rocprofiler-sdk::rocprofiler-static-library
            rocprofiler-sdk::rocprofiler-glog

            rocprofiler-sdk::rocprofiler-hip

    TARGET codeobj-library-test

    TEST_LIST codeobj-library-test_TESTS
